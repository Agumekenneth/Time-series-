# Function to convert notebook to PDF with HTML fallback
from pathlib import Path
import nbformat
from nbconvert import PDFExporter, HTMLExporter

def convert_to_pdf(notebook_path='Assignment_one.ipynb', out_pdf='Assignment_one.pdf'):
    notebook_path = Path(notebook_path)
    if not notebook_path.exists():
        raise FileNotFoundError(f"{notebook_path} not found")

    nb = globals().get('nb') or nbformat.read(str(notebook_path), as_version=4)
    pdf_exporter = globals().get('pdf_exporter') or PDFExporter()

    try:
        body, resources = pdf_exporter.from_notebook_node(nb)
        if isinstance(body, str):
            body = body.encode('utf-8')
        Path(out_pdf).write_bytes(body)
        print(f"Conversion complete: {out_pdf}")
    except Exception as e:
        html_exporter = globals().get('html_exporter') or HTMLExporter()
        body, resources = html_exporter.from_notebook_node(nb)
        html_path = Path(out_pdf).with_suffix('.html')
        html_path.write_text(body, encoding='utf-8')
        print(f"PDF export failed ({e}). Saved HTML fallback: {html_path}")

# call the function

#Function to convert notebook to PDF 
from pathlib import Path
import nbformat
from nbconvert import PDFExporter, HTMLExporter

def convert_notebook_to_pdf(notebook_file='Assignment_one.ipynb', pdf_file='Assignment_one.pdf'):
    """
    Converts a Jupyter notebook to PDF. 
    If PDF fails (e.g., missing LaTeX/Pandoc), it saves an HTML file instead.
    """
    notebook_path = Path(notebook_file)
    if not notebook_path.exists():
        print(f"Notebook not found: {notebook_file}")
        return

    # Read the notebook
    nb = nbformat.read(str(notebook_path), as_version=4)

    # Try PDF conversion
    pdf_exporter = PDFExporter()
    try:
        pdf_data, _ = pdf_exporter.from_notebook_node(nb)
        if isinstance(pdf_data, str):
            pdf_data = pdf_data.encode('utf-8')
        Path(pdf_file).write_bytes(pdf_data)
        print(f" PDF created successfully: {pdf_file}")

    except Exception as e:
        print(f" PDF conversion failed ({e}). Using HTML fallback.")
        # Export as HTML instead
        html_exporter = HTMLExporter()
        html_data, _ = html_exporter.from_notebook_node(nb)
        html_file = Path(pdf_file).with_suffix('.html')
        html_file.write_text(html_data, encoding='utf-8')
        print(f"HTML file saved: {html_file}")

# Calling the function 
convert_notebook_to_pdf()
